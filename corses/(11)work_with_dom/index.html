Обход и поиск элементов

Обход

  • childNodes - Array подобный объект
  • firstChild и lastChild - ссылки на childNodes[0] и childNodes[1]
  • nextSibling и previousSibling - ссылки на соседние елементи узла
  • parentNode - каждый узел DOM содержит ссылку на родительский узел

Обход и поиск элементов

Поиск

  • getElementsByTagName('a')

  • getElementsByClassName('info')

  • getElementsByName('name')

  • querySelectorAll('h1 + p')

  • getElementById('hi')

  • querySelector('p#hi')

<!doctype html>
<html>
  <head>
    <title>My home page</title>
  </head>
  <body>
    <h1>My home page</h1>
    <p id="hi">
        Hi, I'm Denis and 
        it's my home page
    </p>
    <p class="info">
        I also wrote this article!
        Read it
        <a href="http://some-where.com">
            here.
        </a>
    </p>
  </body>
</html>
var parent = document.body,
    anchor = parent.getElementsByTagName('a')[0],
    info = parent.getElementsByClassName('info')[0],
    hi = document.getElementById('hi'),
    qAll = parent.querySelectorAll('h1 + p'),
    q = parent.querySelector('p.info');

console.log(anchor)
// <a> ... </a>

console.log(info)
console.log(q)
// <p class='info'> ... </p>

console.log(hi)
// <p id='hi'> ... </p>

console.log(qAll)
// [<p class='info'> ... </p>]

Изменение DOM

Мы можем:

  • создавать

  • добавлять и перемщать

  • удалять

Изменение DOM

Создание элемента

var nodeName = 'div',
    element = document.createElement(nodeName);

console.log(element);

// <div></div>

console.dir(element);

/*
div
    accessKey: ""
    align: ""
    attributes: NamedNodeMap
    baseURI: (...)
    childElementCount: 0
    classList: DOMTokenList[0]
    className: ""
    ...
*/

Создание элемента происходит через визов метода createElement у document и принимает nodeName в качестве обязательного параметра

Изменение DOM

Добавление и перемещение



var nodeName = 'div',
    element = document.createElement(nodeName),
    parent = document.body;

parent.appendChild(element);

console.log(parent.lastChild);
// <div></div>
parent.insertBefore(parent.lastChild, 
                    parent.firstChild);

console.log(parent.firstChild);
// <div></div>

parent.replaceChild(parent.lastChild, 
                    element);

Добавлять элементы можно двумя методами:

  1. appendChild - добавляет новый элемент в конец;
  2. insertBefore - добавляет новый элемент перед элементом который передается вторым параметром. Также этим методом можно перемещать элементы у родителя;
  3. replaceChild - заменить дочерний элемент.

Изменение DOM

Удаление



var nodeName = 'div',
    element = document.createElement(nodeName),
    parent = document.body;

parent.appendChild(element);

/*
* наш маленький element...
* делал что то важноэ...
* но он нам надоел,
* и мы его люстрируэм.

parent.removeChild(element)

Удаление элементов проходит посредством вызова метода removeChild у parent, который принимает в качестве обязательного параметра сам nodeElemnt который требуется удалить.

Работа с элементами DOM

Работа с атрибутами





var nodeName = 'a',
    anchor = document.createElement(nodeName),
    parent = document.body;

anchor.setAttribute('data-attr',
                    'some attr value');

console.log(anchor.getAttribute('data-attr'));

// 'some attr value'
  • setAttribute - устанавливает атрибут
  • getAttribute - получает значение атрибута

 

Работа с элементами DOM

Кастомные атрибуты



var nodeName = 'a',
    anchor = document.createElement(nodeName),
    parent = document.body;

anchor.setAttribute('data-attr',
                    'some attr value');

console.log(anchor.dataset.attr);
// 'some attr value'

anchor.dataset.attr = 'some new value';

console.log(anchor.dataset.attr);
// 'some new value'
  • "data-" - префикс против конфликтов
  • dataset - объект для работы с кастомными атрибутами
  • dataset - быстрее 

 

Работа с элементами DOM

classList




var nodeName = 'a',
    anchor = document.createElement(nodeName),
    parent = document.body;

anchor.classList.add('specific-class');
anchor.classList.remove('specific-class');
anchor.classList.toggle('specific-class');
anchor.classList.contains('specific-class');

console.log(anchor.classList)
// ['visited', 'red', 'specific-class']
  • className - стар и неудобен
  • classList - объект для работы с классами
