Генератори

Generators *

				function* compute(degree, max) {
					var a = 1;
					for (let i = 0; i < max; i++) {
						yield a;
						a = a * degree;
					}
				}
				let gen = compute(2, 3);
				gen.next(); // { value: 1, done: false }
				gen.next(); // { value: 2, done: false }
				gen.next(); // { value: 4, done: false }
				gen.next(); // { value: undefined, done: true }

Generators *

				function* seq(a) {	
					yield a + 5;	
					let b = yield null;	
					return a + b;	
				}	
				let gen = seq(5);	
				gen.next().value; // 10	
				gen.next().value; // null	
				gen.next(8).value; // 13

Класи

Класи в ES 6

				class Person {
					constructor(name) {
						this.name = name;
					}
					describe() {
						return "Person called " + this.name;
					}
				}
				class Employee extends Person {
					constructor(name, title) {
						super.constructor(name);
						this.title = title;
					}
					describe() {
						return super.describe() + " (" + this.title + ")";
					}
				}
