Подключение скриптов

Подключение с помощью тега <script>

  • Встроенный скрипт (embedded)
  • Внешний скрипт (external)

Встроенный

index.html 
				<script>
					alert('Hello, world!');
				</script>

Внешний

index.html 
				<script src="script.js"></script>
script.js 
				alert('Hello, world!');

Размещение тега script

Интуиция подсказывает, что нужно размещать в теге head

			<head>
				<link rel="stylesheet" href="styles.css">
				<script src="script.js"></script>
			</head>
			<body>
				<!-- Rendering content -->
			</body>

Размещение тега script

Хорошая практика - подключать script в конце тега body

			<head>
				<link rel="stylesheet" href="styles.css">
			</head>
			<body>
				<!-- Rendering content -->
				<script src="script.js"></script>
			</body>

Структура кода

JavaScript код - набор команд, которые должен выполнить браузер.

Такие команды называются инструкциями (statements).

				statement;
				statement;
				statement;

Переменные

Переменная - именованная ячейка памяти.

Объявляется с помощью ключевого слова var и идентификатора.

				var myVar;

Идентификаторы

  • Длина идентификатора – один и больше символов
  • Символы: буква, цифра, подчеркивание (_) или доллар ($)
  • Цифра не может быть первой
  • По соглашению пишутся в стиле camelCase
  • Ключевые и зарезервированные слова нельзя использовать в качестве идентификаторов

Присваивание

				var myVar; // объявление
				myVar = 'my var'; // присваивание

Инициализация

				var myVar = 'my var';

Инициализация лучше

Плохо

				var myVar; // объявление
				myVar = 'my var'; // присваивание

Хорошо

				var myVar = 'my var'; // инициализация

Объявление нескольких переменных

				var name = 'Артем';
				var age = 25;

Единственная инструкция var

				var name = 'Артем',
					 age = 25;

Единственная инструкция var - лучше

Плохо

				var name = 'Артем';
				var age = 25;

Хорошо

				var name = 'Артем',
					 age = 25;

Область видимости

Область видимости переменной - функция в которой она объявлена.

			function foo(){
				var bar = 1;
				// do smth with bar
			}
			foo();
				
			console.log(bar); Uncaught ReferenceError: bar is not defined

Нету блочной области видимости

			{
				var foo = 'bar';
			}
			 
			console.log(foo); // bar

NaN

			"a" * 3 // NaN
			NaN * 3 // NaN

Функция Number

Конвертирует значение в число

			Number("10") // 10
			Number("10.11") // 10.11
			Number("10a") // NaN
			+"10" // 10

Функция parseInt

			parseInt("10px", 10) // 10
			parseInt("px10", 10) // NaN
			parseInt("010") // в некоторых браузерах 8
			parseInt("010", 10) // во всех 10

Динамическая типизация

Тип переменной определяется в момент присваивания значения

				var myVar = 1;
				myVar = "foo";
				myVar = true;

Оператор typeof

Определяет тип переменной

			typeof 1; // "number"
			typeof "some string"; // "string"
			typeof true; // "boolean"
			typeof undefined; // "undefined"
			typeof function(){}; // "function"

Для всего остального typeof возвращает "object", в том числе и для null.

Литералы

			var num = 1,
				str = "string",
				bool = true;

Конструкторы встроенных типов

			var num = new Number(1),
				str = new String("string"),
				bool = new Boolean(true);

Отличие

			var primitiveString = "bar",
				objectString = new String("bar");
			 
			typeof primitiveString; // "string"
			typeof objectString; // "object"

Многие думают, что в JS - все объекты

			var primitiveString = "bar";
			primitiveString.length; // 3

Составная инструкция (block statement)

Позволяет объеденять несколько инструкций в одну.

				{
					statement;
					statement;
					statement;
				}

Лучше всегда указывать фигурные скобки

			if(выражение){
				инструкция1; // statement1
				инструкция2; // statement2
			}else{
				инструкция3; // statement3
			}

Тернарный оператор

			условие ? выражение1 : выражение2
		
			var msg = isTrue ? "Истина" : "Ложь";
			console.log(msg);

Вложенный if else

Плохо

			if(value === 0){
				console.log("Ноль");
			}else{
				if(value === 1){
					console.log("Один");
				}else{
					console.log("Другое");
				}
			}

Вложенный if else

Лучше

			if(value === 0){
				console.log("Ноль");
			}else if(value === 1){
				console.log("Один");
			}else{
				console.log("Другое");
			}

switch case

			switch(value){
				case 0:
					console.log("Ноль");
					break;
				case 1:
					console.log("Один");
					break;
				default:
					console.log("Другое");
			}

Циклы

  • С предусловием
    • while
    • for
  • С постусловием
    • do while

while

			while(выражение)
				инструкция; // statement

Пример

			var i = 0;
			while(i < 3){
				console.log(i++);
			}
			// 0, 1, 2

do while

			var i = 3;
			do{
				console.log(i);
				i++;
			}while(i < 3)
			// 3

for

			for(инициализация; проверка; инкремент)
				инструкция; // statement

Пример

			for(i = 0; i < 3; i++){
				console.log(i);
			}
			// 0, 1, 2

break

			for(i = 0; i < 3; i++){
				if(i === 1){
					break;
				}
				console.log(i);
			}
			// 0

continue

			for(i = 0; i < 3; i++){
				if(i === 1){
					continue;
				}
				console.log(i);
			}
			// 0, 2
