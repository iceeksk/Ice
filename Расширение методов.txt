Function.prototype.method = function (name, func) {
    this.prototype[name] = func;
    return this;
};

Array.method('суммаћасива', function () {
	return this.reduce(function(sum, current) {
 		 return sum + current;
		}, 0);
});

Array.method('добћасива', function () {
		return this.reduce(function(sum, current) {
 		 return sum * current;
		});
});
